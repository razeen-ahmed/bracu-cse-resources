
#include <stdio.h>

int pernum(int num){

    int sum = 0;
    //printf("helooooooooo");
    for (int i=1; i<num;i++){

        if (num%i==0){

            sum += i;

        }
    }
    //printf("%d",&sum);

    if (sum == num){

        return 1;

    }
    else{

        return 0;
    }

};

int main(){

    int start;
    int end;

        get = pernum(i);

}
